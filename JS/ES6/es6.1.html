# 箭头函数

箭头是使用=>语法的函数简写。它们在语法上类似于c#、Java 8和CoffeeScript中的相关特性。它们同时支持表达式和语句主体。与函数不同，箭头的词法与它们周围的代码相同。如果箭头在另一个函数中，它共享其父函数的“arguments”变量。

// 表达式本体
var odds = evens.map(v => v + 1);
var nums = evens.map((v, i) => v + i);

// 声明式
nums.forEach(v => {
  if (v % 5 === 0)
    fives.push(v);
});

// 词法
var bob = {
  _name: "Bob",
  _friends: [],
  printFriends() {
    this._friends.forEach(f =>
      console.log(this._name + " knows " + f));
  }
};

// 词法参数
function square() {
  let example = () => {
    let numbers = [];
    for (let number of arguments) {
      numbers.push(number * number);
    }

    return numbers;
  };

  return example();
}

square(2, 4, 7.5, 8, 11.5, 21); // returns: [4, 16, 56.25, 64, 132.25, 441]
