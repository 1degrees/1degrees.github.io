# 职责链模式

使多个对象都有机会处理请求，从而避免请求的发送者和接收者之间 的耦合关系，将这些对象连成一条链，并沿着这条链传递该请求，直到有一个对象处理它为止。
职责链模式的名字非常形象，一系列可能会处理请求的对象被连接成一条链，请求在这些对 象之间依次传递，直到遇到一个可以处理它的对象，我们把这些对象称为链中的节点，如图 13-1 所示。

# 1. 职责链模式应用

现有任务 A、B、C、D, 要求
第一个任务链，A任务执行完执行C、执行完成执行D
第二个任务链，B任务执行完执行A、执行完成执行C

# 1.1 同步职责链

/* 定义A,B,C,D 任务 */
taskA = function(){
    console.log('A任务执行完成')
    return 'next'
}

taskB = function(){
    console.log('B任务执行完成')
    return 'next'
}

taskC = function(){
    console.log('C任务执行完成')
    return 'next'
}

taskD = function(){
    console.log('D任务执行完成')
    return 'next'
}

/* 定义一个链式结构类 */
class Chain{ 
    fn = function(){}; 
    nextTask = null;
    constructor(fn){
        this.fn = fn;
    }
    setNextTask(next){
        return this.nextTask = next;
    };
    runTask(...parmas){
        let cb = this.fn;
        let rs = cb.apply(this, parmas);
        if(rs==='next' && this.nextTask) {
            this.nextTask.runTask.apply(this.nextTask, parmas);
        }
        return rs
    };
}

var cTaskA = new Chain(taskA)
var cTaskB = new Chain(taskB)
var cTaskC = new Chain(taskC)
var cTaskD = new Chain(taskD)

//第一条任务链
cTaskA.setNextTask(cTaskC).setNextTask(cTaskD)
cTaskA.runTask();

var cTaskA = new Chain(taskA)
var cTaskB = new Chain(taskB)
var cTaskC = new Chain(taskC)
var cTaskD = new Chain(taskD)
//第二条任务链
cTaskB.setNextTask(cTaskA).setNextTask(cTaskC)
cTaskB.runTask();

# 1.2 异步步职责链

现有任务 A、B、C、D任务为异步,实现如下

/* 定义A,B,C,D 任务 */
taskA = function(){
    return new Promise((resolve, reject) => {
        setTimeout(()=>{
            console.log('A任务执行完成')
            resolve('next')
        }, 1000)
    })
}

taskB = function(){
    return new Promise((resolve, reject) => {
        setTimeout(()=>{
            console.log('B任务执行完成')
            resolve('next')
        }, 2000)
    })
}

taskC = function(){
    console.log('C任务执行完成')
    return Promise.resolve('next')
}

taskD = function(){
    console.log('D任务执行完成')
    return Promise.resolve('next')
}

/* 定义一个链式结构类 */
class Chain{ 
    fn = function(){}; 
    nextTask = null;
    constructor(fn){
        this.fn = fn;
    }
    setNextTask(next){
        return this.nextTask = next;
    };
    runTask(...parmas){
        let cb = this.fn;
        cb(parmas).then(rs => {
            if(rs==='next' && this.nextTask) {
                this.runNextTask(parmas);
            }
        });
    };
    runNextTask(parmas) {
        this.nextTask.runTask.apply(this.nextTask, parmas);
    }
}

var cTaskA = new Chain(taskA)
var cTaskB = new Chain(taskB)
var cTaskC = new Chain(taskC)
var cTaskD = new Chain(taskD)

//第一条任务链
cTaskA.setNextTask(cTaskC).setNextTask(cTaskD)
cTaskA.runTask();

var cTaskA = new Chain(taskA)
var cTaskB = new Chain(taskB)
var cTaskC = new Chain(taskC)
var cTaskD = new Chain(taskD)
//第二条任务链
cTaskB.setNextTask(cTaskA).setNextTask(cTaskC)
cTaskB.runTask();

# 1.3 函数AOP方式实现职责链

/* 定义A,B,C,D 任务 */
taskA = function(){
    console.log('A任务执行完成')
    return 'next'
}

taskB = function(){
    console.log('B任务执行完成')
    return 'next'
}

taskC = function(){
    console.log('C任务执行完成')
    return 'next'
}

taskD = function(){
    console.log('D任务执行完成')
    return 'next'
}

/* 定义一个链式结构类 */
Function.prototype.after = function(fn){
    let _self = this;
    return function(){
        let rs = _self.apply(_self, arguments);
        fn(arguments);
        return rs
    }
}
var chain1 = taskA.after(taskC).after(taskD)
chain1();

var chain2 = taskB.after(taskA).after(taskC)
chain2();

# 2. 小结

在 JavaScript 开发中，职责链模式是最容易被忽视的模式之一。实际上只要运用得当，职责 链模式可以很好地帮助我们管理代码，降低发起请求的对象和处理请求的对象之间的耦合性。职 责链中的节点数量和顺序是可以自由变化的，我们可以在运行时决定链中包含哪些节点。 无论是作用域链、原型链，还是 DOM 节点中的事件冒泡，我们都能从中找到职责链模式的 影子。职责链模式还可以和组合模式结合在一起，用来连接部件和父部件，或是提高组合对象的 效率。学会使用职责链模式，相信在以后的代码编写中，将会对你大有裨益
